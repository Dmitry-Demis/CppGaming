    Шаблоны с несколькими параметрами типа

    Шаблон с одним параметром T требует, чтобы оба аргумента были одного типа. Если нужно принимать аргументы разных типов — используют несколько параметров типа.

    Проблема одного T

    C++ — один T не работает с разными типами
    template <typename T>
T max(T x, T y) { return (x < y) ? y : x; }

int main()
{
    max(2, 3.5);   // compile error: T = int или double? Неоднозначно!
    max(2, 3);     // OK: T = int
    max(2.0, 3.5); // OK: T = double
    return 0;
}

    Два параметра типа: T и U

    C++ — шаблон с T и U
    #include <iostream>

template <typename T, typename U>
auto max(T x, U y) // auto: компилятор выводит тип возврата
{
    return (x < y) ? y : x;
}

int main()
{
    std::cout << max(2, 3.5)   << '\n'; // T=int, U=double → 3.5
    std::cout << max(5L, 3)    << '\n'; // T=long, U=int → 5
    std::cout << max(1.5f, 2)  << '\n'; // T=float, U=int → 2
    return 0;
}

    Сокращённый синтаксис C++20: auto-параметры

    C++20 — abbreviated function template
    // C++20: auto-параметры — каждый auto становится независимым T
auto max(auto x, auto y)
{
    return (x < y) ? y : x;
}

// Эквивалентно:
// template <typename T, typename U>
// auto max(T x, U y) { return (x < y) ? y : x; }

// Но если нужно, чтобы оба параметра были ОДНОГО типа — используйте явный T:
template <typename T>
T sum(T x, T y) { return x + y; } // sum(1, 2.5) — ошибка, что и нужно

    Перегрузка шаблонов

    C++ — перегрузка шаблонов функций
    #include <iostream>

// Два одинаковых типа
template <typename T>
auto add(T x, T y) { return x + y; }

// Два разных типа
template <typename T, typename U>
auto add(T x, U y) { return x + y; }

// Три параметра
template <typename T, typename U, typename V>
auto add(T x, U y, V z) { return x + y + z; }

int main()
{
    std::cout << add(1.2, 3.4)    << '\n'; // add<double> (одинаковые)
    std::cout << add(5.6, 7)      << '\n'; // add<double,int> (разные)
    std::cout << add(8, 9, 10)    << '\n'; // add<int,int,int> (три)
    return 0;
}
