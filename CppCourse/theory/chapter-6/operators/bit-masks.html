    Битовые маски

    Битовая маска — это заранее определённый набор битов, который указывает, какие именно биты нужно изменить. Маска блокирует доступ к ненужным битам и открывает доступ к нужным — как малярная лента при покраске.

    Определение масок

    Самый читаемый способ — бинарные литералы C++14:

    C++ — определение битовых масок 
    #include <cstdint>

// C++14: бинарные литералы
constexpr std::uint8_t mask0{ 0b0000'0001 }; // бит 0
constexpr std::uint8_t mask1{ 0b0000'0010 }; // бит 1
constexpr std::uint8_t mask2{ 0b0000'0100 }; // бит 2
// ...

// Альтернатива: сдвиг единицы (работает в C++11 и ранее)
constexpr std::uint8_t bit0{ 1 << 0 }; // 0000 0001
constexpr std::uint8_t bit1{ 1 << 1 }; // 0000 0010
constexpr std::uint8_t bit2{ 1 << 2 }; // 0000 0100

    Проверка бита

    Используем побитовое И (&) с маской:

    C++ — проверка бита 
    #include <cstdint>
#include <iostream>

int main()
{
    constexpr std::uint8_t mask0{ 0b0000'0001 };
    constexpr std::uint8_t mask1{ 0b0000'0010 };

    std::uint8_t flags{ 0b0000'0101 }; // биты 0 и 2 установлены

    std::cout << "бит 0: " << (static_cast<bool>(flags & mask0) ? "вкл\n" : "выкл\n"); // вкл
    std::cout << "бит 1: " << (static_cast<bool>(flags & mask1) ? "вкл\n" : "выкл\n"); // выкл

    return 0;
}

    Установка, сброс и инверсия

    Шпаргалка по операциям с битами
    Операция Код
    Проверить битif (flags & mask)
    Установить битflags |= mask
    Установить несколькоflags |= (mask4 | mask5)
    Сбросить битflags &= ~mask
    Сбросить несколькоflags &= ~(mask4 | mask5)
    Инвертировать битflags ^= mask
    Инвертировать несколькоflags ^= (mask4 | mask5)
    C++ — установка, сброс, инверсия 
    #include <cstdint>
#include <iostream>

int main()
{
    constexpr std::uint8_t isHappy    { 1 << 3 }; // 0000 1000
    constexpr std::uint8_t isLaughing { 1 << 4 }; // 0001 0000

    std::uint8_t me{};
    me |= (isHappy | isLaughing); // установить оба флага
    me &= ~isLaughing;            // сбросить флаг смеха

    std::cout << std::boolalpha;
    std::cout << "Счастлив: "  << static_cast<bool>(me & isHappy)    << '\n'; // true
    std::cout << "Смеётся: "   << static_cast<bool>(me & isLaughing) << '\n'; // false

    return 0;
}

    Пример: разбор RGBA-цвета

    Цвет пикселя часто хранится как 32-битное число, где каждые 8 бит — один канал (R, G, B, A). Маски позволяют извлечь каждый канал:

    C++ — извлечение RGBA-каналов 
    #include <cstdint>
#include <iostream>

int main()
{
    constexpr std::uint32_t redBits   { 0xFF000000 };
    constexpr std::uint32_t greenBits { 0x00FF0000 };
    constexpr std::uint32_t blueBits  { 0x0000FF00 };
    constexpr std::uint32_t alphaBits { 0x000000FF };

    std::uint32_t pixel { 0xFF7F3300 };

    const std::uint8_t red   { static_cast<std::uint8_t>((pixel & redBits)   >> 24) };
    const std::uint8_t green { static_cast<std::uint8_t>((pixel & greenBits) >> 16) };
    const std::uint8_t blue  { static_cast<std::uint8_t>((pixel & blueBits)  >>  8) };
    const std::uint8_t alpha { static_cast<std::uint8_t>(pixel & alphaBits) };

    std::cout << std::hex;
    std::cout << "R=" << static_cast<int>(red)
              << " G=" << static_cast<int>(green)
              << " B=" << static_cast<int>(blue)
              << " A=" << static_cast<int>(alpha) << '\n';
    // R=ff G=7f B=33 A=0

    return 0;
}
