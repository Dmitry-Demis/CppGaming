    Сравнение строк

    Строки сравниваются лексикографически — посимвольно, по числовым кодам символов. Регистр имеет значение: 'A' (65) < 'a' (97).

    C++ — операторы сравнения строк
    #include <iostream>
#include <string>

int main()
{
    std::string s1 { "hello" };
    std::string s2 { "Hello" };
    std::string s3 { "hello" };

    std::cout << (s1 == s3) << '\n'; // 1 (true): равны
    std::cout << (s1 == s2) << '\n'; // 0 (false): регистр разный!
    std::cout << (s1 > s2)  << '\n'; // 1 (true): 'h' > 'H' по ASCII

    // Сортировка массива строк:
    std::string names[] { "Tom", "Alice", "Sam", "Bob" };
    // Пузырьковая сортировка:
    for (int i = 0; i < 3; ++i)
        for (int j = 0; j < 3 - i; ++j)
            if (names[j] > names[j+1])
                std::swap(names[j], names[j+1]);

    for (const auto& n : names)
        std::cout << n << '\n'; // Alice Bob Sam Tom

    return 0;
}

    Операции с символами (<cctype>)

    C++ — функции для работы с символами
    #include <cctype>
#include <iostream>
#include <string>

int main()
{
    char c { 'A' };
    std::cout << std::isupper(c) << '\n'; // 1: заглавная
    std::cout << std::islower(c) << '\n'; // 0: не строчная
    std::cout << std::isalpha(c) << '\n'; // 1: буква
    std::cout << std::isdigit('5') << '\n'; // 1: цифра
    std::cout << std::isspace(' ') << '\n'; // 1: пробел

    // Преобразование регистра:
    std::cout << (char)std::tolower('A') << '\n'; // a
    std::cout << (char)std::toupper('z') << '\n'; // Z

    return 0;
}

    Живое применение: сравнение без учёта регистра

    C++ — сравнение строк без учёта регистра
    #include <cctype>
#include <string>
#include <iostream>

bool equalsIgnoreCase(const std::string& a, const std::string& b)
{
    if (a.length() != b.length()) return false;
    for (size_t i {}; i < a.length(); ++i)
        if (std::tolower(a[i]) != std::tolower(b[i]))
            return false;
    return true;
}

int main()
{
    std::cout << equalsIgnoreCase("HELLO", "hello") << '\n'; // 1
    std::cout << equalsIgnoreCase("Hello", "World") << '\n'; // 0
    return 0;
}

    Живое применение: извлечение цифр из строки

    C++ — извлечение номера телефона
    #include <cctype>
#include <string>
#include <iostream>

int main()
{
    std::string text { "Phone: +1(234)456-78-90" };
    std::string digits;

    for (char c : text)
        if (std::isdigit(c))
            digits += c;

    std::cout << digits << '\n'; // 12344567890
    return 0;
}
