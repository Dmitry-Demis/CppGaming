    Практика: программа подсчёта слов

    Объединим всё, что изучили о строках, в одном практическом примере: программа, которая разбивает текст на слова и подсчитывает их. Это классическая задача, демонстрирующая работу с find_first_of, find_first_not_of, substr и std::vector.

    Алгоритм

    Версия 1: подсчёт слов

    C++ — подсчёт слов в строке 
    #include <iostream>
#include <string>

int countWords(const std::string& text)
{
    const std::string separators { " ,;:.\"!?'\n\t" };
    int count { 0 };

    std::size_t start { text.find_first_not_of(separators) };
    while (start != std::string::npos)
    {
        ++count;
        std::size_t end { text.find_first_of(separators, start + 1) };
        if (end == std::string::npos)
            end = text.length();
        start = text.find_first_not_of(separators, end + 1);
    }
    return count;
}

int main()
{
    std::string text { "When in Rome, do as the Romans do." };
    std::cout << "Слов: " << countWords(text) << '\n'; // 8

    return 0;
}

    Версия 2: извлечение всех слов

    C++ — разбивка на слова 
    #include <iostream>
#include <string>
#include <vector>

std::vector<std::string> splitWords(const std::string& text)
{
    const std::string separators { " ,;:.\"!?'\n\t" };
    std::vector<std::string> words;

    std::size_t start { text.find_first_not_of(separators) };
    while (start != std::string::npos)
    {
        std::size_t end { text.find_first_of(separators, start + 1) };
        if (end == std::string::npos)
            end = text.length();

        words.push_back(text.substr(start, end - start));
        start = text.find_first_not_of(separators, end + 1);
    }
    return words;
}

int main()
{
    std::string text { "When in Rome, do as the Romans do." };
    auto words { splitWords(text) };

    std::cout << "Найдено слов: " << words.size() << '\n';
    for (const auto& w : words)
        std::cout << "  " << w << '\n';

    return 0;
}

    Версия 3: с std::string_view (эффективная)

    Используя std::string_view, можно избежать копирования подстрок. Это особенно важно для больших текстов.

    C++ — без копирования через string_view 
    #include <iostream>
#include <string>
#include <string_view>
#include <vector>

std::vector<std::string_view> splitView(std::string_view text)
{
    const std::string_view separators { " ,;:.\"!?'\n\t" };
    std::vector<std::string_view> words;

    std::size_t start { text.find_first_not_of(separators) };
    while (start != std::string_view::npos)
    {
        std::size_t end { text.find_first_of(separators, start + 1) };
        if (end == std::string_view::npos)
            end = text.length();

        words.push_back(text.substr(start, end - start)); // без копирования!
        start = text.find_first_not_of(separators, end + 1);
    }
    return words;
}

int main()
{
    std::string text { "Hello, world! How are you?" };
    auto words { splitView(text) };

    for (const auto w : words)
        std::cout << w << '\n';

    return 0;
}
    Что мы использовали

    find_first_not_of — найти первый символ, не входящий в набор
    find_first_of — найти первый символ из набора
    substr — извлечь подстроку
    std::string::npos — "не найдено"
    std::vector::push_back — добавить слово в список
