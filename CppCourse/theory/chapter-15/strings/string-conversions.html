    Строки

    Преобразования типов и строки

    Часто нужно преобразовать число в строку или строку в число. Стандартная библиотека предоставляет удобные функции для этого.

    Число → строка: std::to_string

    C++ — to_string
    #include <string>
#include <iostream>

int main()
{
    int age { 25 };
    double pi { 3.14159 };

    std::string msg1 { "Возраст: " + std::to_string(age) };
    std::string msg2 { "Pi = " + std::to_string(pi) };

    std::cout << msg1 << '\n'; // Возраст: 25
    std::cout << msg2 << '\n'; // Pi = 3.141590

    return 0;
}

    Строка → число: stoi, stod и другие

    C++ — stoi, stod, stol
    #include <string>
#include <iostream>

int main()
{
    std::string numStr { "42" };
    std::string piStr  { "3.14" };

    int n    { std::stoi(numStr) };  // string → int
    double d { std::stod(piStr) };   // string → double
    long l   { std::stol("1000") };  // string → long

    std::cout << n << '\n'; // 42
    std::cout << d << '\n'; // 3.14

    // Обработка ошибок:
    try {
        int bad { std::stoi("abc") }; // бросает std::invalid_argument
    } catch (const std::exception& e) {
        std::cout << "Ошибка: " << e.what() << '\n';
    }

    return 0;
}

    Строка → C-строка: c_str() и data()

    C++ — c_str и data
    #include <string>
#include <cstdio> // для printf

int main()
{
    std::string msg { "hello" };

    // c_str(): указатель на const char* (для C API)
    const char* cptr = msg.c_str();
    printf("%s\n", cptr); // hello

    // data(): указатель на char* (C++17, не const если строка не const)
    char* ptr = msg.data();
    ptr[0] = 'H'; // изменяем через data()
    printf("%s\n", ptr); // Hello

    return 0;
}
