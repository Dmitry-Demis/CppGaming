    Строки

    Введение в std::string

    std::string — основной тип для работы со строками в C++. Он хранит последовательность символов, автоматически управляет памятью и предоставляет богатый набор методов. Подключается через #include <string>.

    Создание строк

    C++ — способы создания std::string
    #include <string>
#include <iostream>

int main()
{
    std::string empty;                          // пустая строка
    std::string hello { "Hello, world!" };      // из литерала
    std::string copy { hello };                 // копия другой строки
    std::string repeated(4, 'd');               // "dddd" — 4 символа 'd'
    std::string part { "hello world", 5 };      // первые 5 символов: "hello"
    std::string sub { "hello world", 6, 5 };    // 5 символов с позиции 6: "world"

    std::cout << hello << '\n';    // Hello, world!
    std::cout << repeated << '\n'; // dddd
    std::cout << sub << '\n';      // world
    return 0;
}

    Ввод и вывод

    C++ — ввод строки
    #include <iostream>
#include <string>

int main()
{
    std::string name;

    // cin >> читает до первого пробела
    std::cout << "Введите имя: ";
    std::cin >> name;
    std::cout << "Привет, " << name << "!\n";

    std::cin.ignore(); // очищаем буфер после cin >>

    // getline читает всю строку включая пробелы
    std::string fullName;
    std::cout << "Введите полное имя: ";
    std::getline(std::cin, fullName);
    std::cout << "Полное имя: " << fullName << '\n';

    return 0;
}

    Объединение строк

    C++ — конкатенация
    #include <string>
#include <iostream>

int main()
{
    std::string hello { "hello" };
    std::string world { "world" };
    std::string msg { hello + " " + world }; // "hello world"
    std::cout << msg << '\n';

    // Нельзя складывать два литерала напрямую:
    // std::string bad { "hello " + "world" }; // ОШИБКА!

    // Суффикс s делает литерал std::string:
    using namespace std::string_literals;
    std::string ok { "hello "s + "world"s }; // OK
    std::cout << ok << '\n';

    return 0;
}

    Raw-литералы

    C++ — raw-строки (без экранирования)
    // Обычная строка с экранированием:
std::string path1 { "C:\\Users\\Tom\\file.txt" };

// Raw-литерал: R"(...)" — всё внутри буквально
std::string path2 { R"(C:\Users\Tom\file.txt)" };

// Многострочный raw-литерал:
std::string json { R"({
    "name": "Tom",
    "age": 25
})" };

std::cout << path2 << '\n'; // C:\Users\Tom\file.txt
