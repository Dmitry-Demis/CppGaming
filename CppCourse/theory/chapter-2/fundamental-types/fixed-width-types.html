    Типы с фиксированным размером

    Стандарт C++ не гарантирует точных размеров для int, long и других базовых типов. Когда точный размер критичен — сетевые протоколы, бинарные форматы, аппаратные регистры — используйте типы из <cstdint>.

    Заголовок <cstdint>

    [cstdint.syn] — ISO/IEC 14882:2020 The header <cstdint> defines all functions, types, and macros the same as the C standard library header <stdint.h>.

    Типы делятся на четыре категории:

    Тип Размер Знаковость Диапазон / max Применение
    — Точный размер (exact-width) —
    int8_t8 битsigned ?128 … 127 Аппаратные регистры, компактные флаги
    uint8_t8 битunsigned 0 … 255 Байт данных, пиксель, байт буфера
    int16_t16 битsigned ?32 768 … 32 767 Аудио-сэмпл PCM 16-bit
    uint16_t16 битunsigned 0 … 65 535 Порт TCP/UDP, Unicode codepoint (BMP)
    int32_t32 битаsigned ±2.1?10? IPv4-адрес, цвет ARGB
    uint32_t32 битаunsigned 0 … 4.3?10? CRC32, хеш-значения
    int64_t64 битаsigned ±9.2?10?? Unix timestamp, ID записей в БД
    uint64_t64 битаunsigned 0 … 18.4?10?? 64-битная маска, SHA-256 слово
    — Минимальный размер (least-width) —
    int_least8_t?8 битsigned Гарантирует хотя бы 8 бит; может быть шире
    — Быстрейший тип заданного минимума (fast) —
    int_fast8_t?8 битsigned Самый быстрый тип, вмещающий ?8 бит (часто 32 или 64)
    — Pointer-sized типы —
    intptr_t?sizeof(void*)signed платформенный Хранение указателя как числа
    uintptr_t?sizeof(void*)unsigned платформенный Адресная арифметика
    std::size_t?sizeof(void*)unsigned 0 … platform max Размеры объектов, индексы контейнеров
    std::ptrdiff_t?sizeof(void*)signed платформенный Разность двух указателей
    C++11/17/20 — <cstdint> на практике 
    #include <iostream>
#include <cstdint>
#include <format>   // C++20

// Бинарный заголовок BMP: каждое поле — точный размер
#pragma pack(push, 1)
struct BitmapFileHeader {
    uint16_t signature{0x4D42};   // "BM"
    uint32_t fileSize{};
    uint32_t reserved{};
    uint32_t dataOffset{};
};
#pragma pack(pop)

int main() {
    // C++11: static_assert проверяет размер на этапе компиляции
    static_assert(sizeof(BitmapFileHeader) == 14, "BMP header must be 14 bytes");
    static_assert(sizeof(int32_t)  == 4);
    static_assert(sizeof(uint64_t) == 8);

    // C++14: digit separators
    uint64_t timestamp{1'710'000'000'000ULL};  // ms с Unix epoch

    // C++17: structured bindings (пример с парой)
    auto [lo, hi] = std::pair{uint32_t{0xDEAD'BEEFu}, uint32_t{0xCAFE'BABEu}};

    // C++20: std::format с hex
    std::cout << std::format("lo = {:#010x}, hi = {:#010x}\n", lo, hi);
    std::cout << std::format("timestamp = {}\n", timestamp);

    // size_t для контейнеров и sizeof
    std::size_t bufSize{1024};
    std::cout << std::format("bufSize = {} bytes\n", bufSize);

    return 0;
}

    std::numeric_limits — интроспекция типов

    Шаблон std::numeric_limits<T> из <limits> предоставляет compile-time константы для любого числового типа. Это переносимый способ узнать диапазон без «магических чисел».

    ЧленТип возвратаОписание
    min()TМинимальное значение (для float — минимум по модулю > 0)
    max()TМаксимальное конечное значение
    lowest()TНаименьшее конечное значение (= min() для целых, -max() для float)
    digitsintКоличество значащих бит (без знака)
    is_signedbooltrue для знаковых типов
    is_integerbooltrue для целочисленных типов
    C++11/23 — std::numeric_limits 
    #include <iostream>
#include <limits>
#include <cstdint>
#include <print>    // C++23

template<typename T>
void showType(const char* name) {
    using nl = std::numeric_limits<T>;
    std::print("{:25s}: {:2d}b  signed={:1}  min={:25}  max={}\n",
        name,
        sizeof(T),
        nl::is_signed,
        (long long)nl::min(),
        (unsigned long long)nl::max()
    );
}

int main() {
    showType<int8_t>  ("int8_t");
    showType<uint8_t> ("uint8_t");
    showType<int16_t> ("int16_t");
    showType<int32_t> ("int32_t");
    showType<int64_t> ("int64_t");
    showType<uint64_t>("uint64_t");
    showType<short>   ("short");
    showType<int>     ("int");
    showType<long>    ("long");
    showType<long long>("long long");

    // Compile-time использование:
    constexpr int MAX_SCORE{std::numeric_limits<int>::max()};
    static_assert(MAX_SCORE == 2'147'483'647);

    return 0;
}

    Когда использовать fixed-width типы?

    ? Используйте int32_t / uint8_t когда:
    • Бинарные форматы — BMP, PNG, JPEG, ZIP, ELF
    • Сетевые протоколы — TCP/IP заголовки, HTTP/2 frames
    • Аппаратные регистры — MMIO, USB, I2C, SPI
    • Межплатформенная совместимость — сериализация, IPC
    • Память критична — большие массивы uint8_t vs int
    Используйте int / long long для общего кода

    В обычном бизнес-логике: счётчики, флаги, результаты вычислений — используйте int (или long long для больших значений). Базовые типы более читаемы и часто быстрее на конкретной платформе. Fixed-width типы — для случаев, где размер имеет семантическое значение.

    C++11/17/20 — выбор типа по контексту 
    #include <iostream>
#include <cstdint>
#include <vector>
#include <span>     // C++20

// ? Бинарный парсер: uint8_t/uint32_t — точный размер важен
struct PngChunk {
    uint32_t length{};     // big-endian, 4 байта
    uint32_t type{};       // 4 ASCII-символа
    // ... data ...
    uint32_t crc{};        // CRC32
};

// ? Бизнес-логика: int — читаемо и достаточно
int calculateScore(int base, int bonus, int penalty) {
    return base + bonus - penalty;  // диапазон int более чем достаточен
}

// ? Буфер данных: span + uint8_t
void processBuffer(std::span<const uint8_t> data) {  // C++20: std::span
    std::cout << "Processing " << data.size() << " bytes\n";
}

int main() {
    // ? Правильный тип для размеров
    std::vector<int> v(100);
    for (std::size_t i{0}; i < v.size(); ++i)
        v[i] = static_cast<int>(i * i);

    // ? C++20: range-based с индексом через std::views::enumerate (C++23)
    for (auto& val : v)
        val *= 2;

    std::cout << "Score: " << calculateScore(100, 50, 20) << '\n';
    return 0;
}
