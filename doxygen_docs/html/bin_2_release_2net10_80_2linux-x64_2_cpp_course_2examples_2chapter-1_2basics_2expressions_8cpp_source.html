1// Выражения (expressions)
2// Выражение — любая комбинация литералов, переменных и операторов,
3// которая вычисляется в единственное значение определённого типа.
4
5#include <iostream>
6
7int main()
8{
9 // Предпочитаем brace-initialization (без лишних пробелов внутри скобок)
10 int x{ 5 };
11 int y{ 3 };
12
13 // x + y, x * 2 + y и т.д. — это выражения. Они вычисляются в значение.
14 // Когда мы записываем это значение в переменную через '=' или '{...}',
15 // вся строка становится инструкцией объявления с инициализацией.
16 int a{ x + y }; // Выражение: x + y (результат 8)
17 int b{ x * 2 + y }; // Выражение: x * 2 + y (результат 13)
18 int c{ (x + y) * 2 }; // Выражение: (x + y) * 2 (результат 16)
19
20 std::cout << "x + y = " << a << '\n';
21 std::cout << "x * 2 + y = " << b << '\n';
22 std::cout << "(x + y) * 2 = " << c << '\n';
23
24 // Выражение можно встраивать напрямую — вычисление произойдёт «на лету»
25 // Здесь выражение x * x + y * y не присваивается переменной, а сразу передаётся в поток.
26 std::cout << "Прямо в cout: " << x * x + y * y << '\n';
27
28 // Тип выражения определяется правилами C++ на основе типов операндов,
29 // а не типом переменной, куда (если вообще куда) записывается результат.
30
31 // Пример 1: Один операнд double → всё выражение вычисляется как double
32 double result{ 7.0 / 2 }; // 3.5
33
34 // Пример 2: Оба операнда int → целочисленное деление (дробная часть отбрасывается)
35 // Даже если мы запишем результат в double, выражение 7 / 2 всё равно даст 3.
36 // Это частая ошибка: потеря точности происходит до момента присваивания.
37 double lost_precision{ 7 / 2 }; // 3.0 (неверно ожидалось 3.5)
38 int trunc{ 7 / 2 }; // 3
39
40 std::cout << "7.0 / 2 = " << result << " (корректно)\n";
41 std::cout << "7 / 2 (запись в double) = " << lost_precision << " (потеря точности!)\n";
42 std::cout << "7 / 2 (запись в int) = " << trunc << '\n';
43
44 // 💡 Запомни: выражение ≠ инструкция.
45 // Выражение вычисляется в значение (например, 5, x + y, func()).
46 // Инструкция — это законченное действие программы (часто: выражение + ';').
47 // Понимание этой разницы поможет избежать ошибок с побочными эффектами и порядком вычислений.
48
49 return 0;
50}
