20 public async Task<List<int>> PickQuestionIdsAsync(
21 int userId, string testId, List<int> allQuestionIds, int pick)
22 {
23 var progressMap = (await repo.GetByTestAsync(userId, testId))
24 .ToDictionary(q => q.QuestionId);
25
26 var rng = new Random();
27
28 // Новые — никогда не видели (абсолютный приоритет)
29 var unseen = allQuestionIds
30 .Where(id => !progressMap.ContainsKey(id))
31 .OrderBy(_ => rng.Next())
32 .ToList();
33
34 // Неправильные из последней попытки (IsCorrect=false)
35 var wrong = allQuestionIds
36 .Where(id => progressMap.TryGetValue(id, out var p) && !p.IsCorrect)
37 .OrderBy(_ => rng.Next())
38 .ToList();
39
40 // Видели и ответили правильно — самые давние первыми
41 var seenCorrect = allQuestionIds
42 .Where(id => progressMap.TryGetValue(id, out var p) && p.IsCorrect)
43 .OrderBy(id => progressMap[id].LastSeenAt ?? DateTime.MinValue)
44 .ToList();
45
46 var result = new List<int>();
47
48 // 1. Сначала все unseen — они должны попасть в первую очередь
49 foreach (var id in unseen)
50 {
51 if (result.Count >= pick) break;
52 result.Add(id);
53 }
54
55 // 2. Добираем неправильными, но не более 50% от оставшихся слотов
56 int remaining = pick - result.Count;
57 int wrongCap = remaining > 0 ? (int)Math.Ceiling(remaining * 0.5) : 0;
58 var wrongAdded = new List<int>();
59 foreach (var id in wrong.Take(wrongCap))
60 {
61 if (result.Count >= pick) break;
62 result.Add(id);
63 wrongAdded.Add(id);
64 }
65
66 // 3. Добираем самыми давними правильными
67 foreach (var id in seenCorrect)
68 {
69 if (result.Count >= pick) break;
70 result.Add(id);
71 }
72
73 // 4. Если всё ещё не хватает — добираем оставшимися неправильными
74 foreach (var id in wrong.Except(wrongAdded))
75 {
76 if (result.Count >= pick) break;
77 result.Add(id);
78 }
79
80 // Перемешиваем финальный список
81 for (int i = result.Count - 1; i > 0; i--)
82 {
83 int j = rng.Next(i + 1);
84 (result[i], result[j]) = (result[j], result[i]);
85 }
86
87 return result;
88 }

94 public async Task UpdateProgressAsync(
95 int userId, string testId, string paragraphId,
96 Dictionary<int, bool> answers)
97 {
98 var today = DateTime.UtcNow.Date;
99 var progressMap = (await repo.GetByTestAsync(userId, testId))
100 .ToDictionary(q => q.QuestionId);
101
102 var updates = new List<QuestionProgress>();
103
104 foreach (var (questionId, isCorrect) in answers)
105 {
106 progressMap.TryGetValue(questionId, out var existing);
107
108 int prevStreak = existing?.CorrectStreak ?? 0;
109 int prevInterval = existing?.SrInterval ?? 1;
110
111 int newStreak = isCorrect ? prevStreak + 1 : 0;
112 int newInterval = isCorrect
113 ? NextInterval(prevInterval, true)
114 : 1;
115
116 updates.Add(new QuestionProgress
117 {
118 UserId = userId,
119 TestId = testId,
120 QuestionId = questionId,
121 ParagraphId = paragraphId,
122 IsCorrect = isCorrect,
123 CorrectStreak = newStreak,
124 SrInterval = newInterval,
125 LastSeenAt = DateTime.UtcNow,
126 NextDueAt = today.AddDays(newInterval)
127 });
128 }
129
130 await repo.UpsertBatchAsync(updates);
131 }

137 public async Task<List<ReviewDueInfo>> GetDueParagraphsAsync(int userId)
138 {
139 var today = DateTime.UtcNow.Date;
140 var all = await repo.GetAllAsync(userId);
141
142 return all
143 .Where(q => q.NextDueAt.HasValue && q.NextDueAt.Value.Date <= today)
144 .GroupBy(q => q.ParagraphId)
145 .Select(g => new ReviewDueInfo(
146 g.Key,
147 g.Select(q => q.TestId).Distinct().ToList(),
148 g.Count(q => !q.IsCorrect),
149 g.Count(),
150 g.Min(q => q.NextDueAt!.Value)
151 ))
152 .OrderBy(r => r.OldestDue)
153 .ToList();
154 }
