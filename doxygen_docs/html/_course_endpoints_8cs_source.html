15 // GET /api/examples/{**path}
16 private static IResult GetExample(string path, IWebHostEnvironment env)
17 {
18 if (path.Contains("..")) return Results.BadRequest();
19 var root = Path.GetFullPath(Path.Combine(env.ContentRootPath, "examples"));
20 var filePath = Path.GetFullPath(Path.Combine(root, path));
21 if (!filePath.StartsWith(root)) return Results.BadRequest();
22 if (!File.Exists(filePath)) return Results.NotFound();
23 return Results.Ok(new { path, code = File.ReadAllText(filePath) });
24 }

37 // GET /api/course-structure
38 private static IResult GetCourseStructure(string cppCoursePath, IWebHostEnvironment env)
39 {
40 var metaPath = Path.Combine(cppCoursePath, "course-meta.json");
41 if (!File.Exists(metaPath))
42 return Results.NotFound(new { message = "course-meta.json not found" });
43
44 using var meta = JsonDocument.Parse(File.ReadAllText(metaPath));
45 var testsRoot = Path.Combine(env.ContentRootPath, "tests");
46 var theoryRoot = Path.Combine(cppCoursePath, "theory");
47
48 var chapters = meta.RootElement.GetProperty("chapters").EnumerateArray()
49 .Select(ch => BuildChapter(ch, theoryRoot, testsRoot, env.ContentRootPath))
50 .ToList();
51
52 return Results.Ok(new { chapters });
53 }

55 private static object BuildChapter(JsonElement ch, string theoryRoot, string testsRoot, string contentRoot)
56 {
57 var chapterId = ch.GetProperty("id").GetString()!;
58 var groupId = ch.GetProperty("groupId").GetString()!;
59 var title = ch.GetProperty("title").GetString();
60 var desc = ch.TryGetProperty("description", out var d) ? d.GetString() : null;
61 var chNum = ch.TryGetProperty("number", out var num) ? num.GetInt32() : 0;
62
63 var theoryDir = Path.Combine(theoryRoot, chapterId, groupId);
64 var chTestsDir = Path.Combine(testsRoot, chapterId, groupId);
65 var chExamplesDir = Path.Combine(contentRoot, "examples", chapterId, groupId);
66
67 var paraIds = GetParaIds(ch, theoryDir);
68 var paragraphs = paraIds.Select(id => BuildParagraph(id, theoryDir, chTestsDir, chExamplesDir))
69 .ToList<object>();
70
71 return new { id = chapterId, number = chNum, groupId, title, description = desc, paragraphs };
72 }

74 private static IEnumerable<string> GetParaIds(JsonElement ch, string theoryDir)
75 {
76 if (ch.TryGetProperty("paragraphs", out var metaParas) && metaParas.ValueKind == JsonValueKind.Array)
77 return metaParas.EnumerateArray()
78 .Select(p => p.GetString()!)
79 .Where(id => !string.IsNullOrEmpty(id));
80
81 if (!Directory.Exists(theoryDir)) return Enumerable.Empty<string>();
82 return Directory.GetFiles(theoryDir, "*.html")
83 .Where(f => !Path.GetFileName(f).Equals("index.html", StringComparison.OrdinalIgnoreCase))
84 .OrderBy(f => f)
85 .Select(f => Path.GetFileNameWithoutExtension(f)!);
86 }

88 private static object BuildParagraph(string paraId, string theoryDir, string testsDir, string examplesDir)
89 {
90 var title = GetHtmlTitle(Path.Combine(theoryDir, paraId + ".html")) ?? paraId;
91 var tests = LoadParaFiles(testsDir, paraId, "*.json");
92 var examples = LoadParaFiles(examplesDir, paraId, "*.cpp");
93 return new { id = paraId, title, tests, examples };
94 }

96 // Loads files for a paragraph: {dir}/{paraId}.ext OR {dir}/{paraId}/*.ext
97 private static List<object> LoadParaFiles(string dir, string paraId, string pattern)
98 {
99 var result = new List<object>();
100 var ext = Path.GetExtension(pattern).TrimStart('*');
101
102 var singleFile = Path.Combine(dir, paraId + ext);
103 if (File.Exists(singleFile))
104 {
105 var item = ext == ".json" ? ReadTestMeta(singleFile) : (object)new { id = Path.GetFileNameWithoutExtension(singleFile) };
106 if (item != null) result.Add(item);
107 return result;
108 }
109
110 var folder = Path.Combine(dir, paraId);
111 if (Directory.Exists(folder))
112 foreach (var f in Directory.GetFiles(folder, pattern).OrderBy(x => x))
113 {
114 var item = ext == ".json" ? ReadTestMeta(f) : (object)new { id = Path.GetFileNameWithoutExtension(f) };
115 if (item != null) result.Add(item);
116 }
117
118 return result;
119 }

121 private static object? ReadTestMeta(string filePath)
122 {
123 try
124 {
125 using var doc = JsonDocument.Parse(File.ReadAllText(filePath));
126 var quizId = doc.RootElement.TryGetProperty("quizId", out var qid) ? qid.GetString() : Path.GetFileNameWithoutExtension(filePath);
127 var title = doc.RootElement.TryGetProperty("title", out var tt) ? tt.GetString() : quizId;
128 var type = doc.RootElement.TryGetProperty("type", out var tp) ? tp.GetString() : "mini";
129 return new { quizId, title, type };
130 }
131 catch { return null; }
132 }

134 private static string? GetHtmlTitle(string filePath)
135 {
136 if (!File.Exists(filePath)) return null;
137 var html = File.ReadAllText(filePath);
138 var m = System.Text.RegularExpressions.Regex.Match(html, @"<h1[^>]*>(.*?)</h1>", System.Text.RegularExpressions.RegexOptions.Singleline);
139 if (!m.Success) return null;
140 return System.Text.RegularExpressions.Regex.Replace(m.Groups[1].Value, "<[^>]+>", "").Trim();
141 }
