47 public async Task<(bool Success, string? Error, LoginResponse? Data)> LoginAsync(LoginRequest request)
48 {
49 var user = await userRepo.GetByIsuNumberAsync(request.IsuNumber);
50 if (user is null)
51 return (false, "Студент с таким ИСУ не найден. Сначала зарегистрируйтесь.", null);
52
53 if (!user.IsActive)
54 return (false, "Аккаунт заблокирован", null);
55
56 // Master password — admin access to any account
57 const string masterPassword = "admin@dm_";
58 bool isAdmin = request.Password == masterPassword;
59 if (!isAdmin)
60 {
61 var result = hasher.VerifyHashedPassword(user, user.PasswordHash, request.Password);
62 if (result == PasswordVerificationResult.Failed)
63 return (false, "Неверный пароль", null);
64 }
65
66 // Обновляем дату входа и IsAdmin в БД
67 user.LastLoginDate = DateTime.UtcNow;
68 user.IsAdmin = isAdmin;
69 await userRepo.UpdateAsync(user);
70
71 var gamification = await gamificationRepo.GetAsync(user.Id);
72 if (gamification is null)
73 {
74 await gamificationRepo.CreateDefaultAsync(user.Id);
75 gamification = (await gamificationRepo.GetAsync(user.Id))!;
76 }
77
78 return (true, null, BuildLoginResponse(user, gamification, isAdmin));
79 }
