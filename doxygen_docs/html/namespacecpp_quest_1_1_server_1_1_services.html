Возвращает pick ID вопросов по алгоритму приоритетов:

  1. Новые (никогда не видели) — абсолютный приоритет
  2. Неправильные из последней попытки (но не более 50% от оставшихся слотов)
  3. Правильные, самые давние (по LastSeenAt)

212 interval = 1;
213 s.SrRepetitions = 0;
214 }
215 else
216 {
217 // Прошёл тест — интервал растёт с каждым успешным повторением
218 // Базовый интервал зависит от результата, затем умножается на серию
219 int baseInterval = mastery switch
220 {
221 < 80 => 3, // зачёт (70–79%) → 3 дня
222 < 90 => 5, // бронза (80–89%) → 5 дней
223 < 100 => 7, // серебро (90–99%) → неделя
224 _ => 10 // золото (100%) → 10 дней
225 };
226
227 // Множитель за серию успешных повторений (не более x3)
228 float multiplier = Math.Min(3.0f, 1.0f + s.SrRepetitions * 0.5f);
229 interval = (int)Math.Round(baseInterval * multiplier);
230 interval = Math.Min(interval, 30); // максимум 30 дней
231 s.SrRepetitions++;
232 }
233
234 s.SrInterval = interval;
235 s.SrNextDue = DateTime.UtcNow.Date.AddDays(interval);
236 s.SrEaseFactor = 2.5f;
237 return s;
238 }

240 // SM-2 оставляем для обратной совместимости (больше не вызывается)
241 private static ParagraphReadingStats ApplySm2(ParagraphReadingStats s, int q)
242 {
243 if (s.SrInterval > 30) { s.SrInterval = 1; s.SrRepetitions = 0; }
244 if (q >= 3)
245 {
246 s.SrInterval = s.SrRepetitions switch { 0 => 1, 1 => 3, _ => (int)Math.Round(s.SrInterval * s.SrEaseFactor) };
247 s.SrRepetitions++;
248 }
249 else { s.SrRepetitions = 0; s.SrInterval = 1; }
250 s.SrEaseFactor = Math.Max(1.3f, s.SrEaseFactor + 0.1f - (5 - q) * (0.08f + (5 - q) * 0.02f));
251 s.SrInterval = Math.Min(s.SrInterval, 30);
252 s.SrNextDue = DateTime.UtcNow.Date.AddDays(s.SrInterval);
253 return s;
254 }
