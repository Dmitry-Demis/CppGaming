15 public async Task<QuizResponse?> GetQuizAsync(string quizId)
16 {
17 var file = FindTestFile(quizId);
18 if (file is null) return null;
19
20 var json = await File.ReadAllTextAsync(file);
21 if (string.IsNullOrWhiteSpace(json)) return null;
22
23 JsonDocument doc;
24 try { doc = JsonDocument.Parse(json); }
25 catch { return null; }
26
27 var root = doc.RootElement;
28
29 var questions = root.GetProperty("questions").EnumerateArray().Select(q =>
30 {
31 var type = q.GetProperty("type").GetString()!;
32 var code = q.TryGetProperty("code", out var c) ? c.GetString() : null;
33 var answers = q.TryGetProperty("answers", out var ans)
34 ? ans.EnumerateArray().Select(a => a.GetString()!).ToList()
35 : null;
36 var explanation = q.TryGetProperty("explanation", out var ex) ? ex.GetString() : null;
37 var std = q.TryGetProperty("std", out var stdProp) ? stdProp.GetString() : null;
38
39 List<QuizPairDto>? pairs = null;
40 if (q.TryGetProperty("pairs", out var pairsEl) && pairsEl.ValueKind == JsonValueKind.Array)
41 {
42 pairs = pairsEl.EnumerateArray().Select(p => new QuizPairDto(
43 p.GetProperty("left").GetString()!,
44 p.GetProperty("right").GetString()!
45 )).ToList();
46 }
47
48 // items и categories для типа classify
49 List<string>? items = null;
50 if (q.TryGetProperty("items", out var itemsEl) && itemsEl.ValueKind == JsonValueKind.Array)
51 items = itemsEl.EnumerateArray().Select(x => x.GetString()!).ToList();
52
53 List<string>? categories = null;
54 if (q.TryGetProperty("categories", out var catsEl) && catsEl.ValueKind == JsonValueKind.Array)
55 categories = catsEl.EnumerateArray().Select(x => x.GetString()!).ToList();
56
57 object? correct = null;
58 if (q.TryGetProperty("correct", out var corr))
59 {
60 correct = ParseCorrect(corr);
61 }
62
63 return new QuizQuestionDto(
64 q.GetProperty("id").GetInt32(),
65 type,
66 q.GetProperty("question").GetString()!,
67 code,
68 answers,
69 correct,
70 explanation,
71 std,
72 pairs,
73 items,
74 categories
75 );
76 }).ToList();
77
78 return new QuizResponse(
79 root.TryGetProperty("quizId", out var qid) ? qid.GetString()! :
80 root.TryGetProperty("id", out JsonElement id) ? id.GetString()! : quizId,
81 root.GetProperty("title").GetString()!,
82 root.TryGetProperty("type", out var tp) ? tp.GetString()! : "mini",
83 root.TryGetProperty("passingScore", out var ps) ? ps.GetInt32() : 70,
84 root.TryGetProperty("pick", out var pk) ? pk.GetInt32() : questions.Count,
85 questions
86 );
87 }

89 public async Task<QuizSubmitResponse?> SubmitQuizAsync(string quizId, QuizSubmitRequest request)
90 {
91 var file = FindTestFile(quizId);
92 if (file is null) return null;
93
94 var json = await File.ReadAllTextAsync(file);
95 if (string.IsNullOrWhiteSpace(json)) return null;
96
97 JsonDocument doc;
98 try { doc = JsonDocument.Parse(json); }
99 catch { return null; }
100
101 var root = doc.RootElement;
102 var passingScore = root.TryGetProperty("passingScore", out var ps) ? ps.GetInt32() : 70;
103
104 var results = new List<QuizResultDto>();
105 int earned = 0;
106 int total = 0;
107
108 foreach (var q in root.GetProperty("questions").EnumerateArray())
109 {
110 total++;
111 var id = q.GetProperty("id").GetInt32();
112 var type = q.GetProperty("type").GetString()!;
113 var explanation = q.TryGetProperty("explanation", out var ex) ? ex.GetString()! : "";
114
115 request.Answers.TryGetValue(id.ToString(), out var submitted);
116
117 bool isRight = false;
118 int pts = 0;
119
120 if (type == "single" || type == "code")
121 {
122 var correct = q.GetProperty("correct").GetInt32();
123 isRight = submitted is JsonElement el
124 && el.ValueKind == JsonValueKind.Number
125 && el.GetInt32() == correct;
126 pts = isRight ? 10 : 0;
127 }
128 else if (type == "multiple")
129 {
130 var correct = q.GetProperty("correct").EnumerateArray()
131 .Select(x => x.GetInt32()).OrderBy(x => x).ToList();
132 List<int> userList = [];
133 if (submitted is JsonElement arr && arr.ValueKind == JsonValueKind.Array)
134 userList = arr.EnumerateArray().Select(x => x.GetInt32()).OrderBy(x => x).ToList();
135 isRight = correct.SequenceEqual(userList);
136 var partial = correct.Any(c => userList.Contains(c));
137 pts = isRight ? 10 : partial ? 5 : 0;
138 }
139 else if (type == "fill")
140 {
141 var accepted = q.GetProperty("correct").EnumerateArray()
142 .Select(x => x.GetString()?.ToLower()).ToList();
143 var userVal = submitted is JsonElement sv ? sv.GetString()?.Trim().ToLower() : null;
144 isRight = userVal != null && accepted.Contains(userVal);
145 pts = isRight ? 10 : 0;
146 }
147
148 earned += pts;
149 results.Add(new QuizResultDto(id, isRight, pts, explanation, q.GetProperty("correct")));
150 }
151
152 int maxScore = total * 10;
153 int pct = maxScore > 0 ? (int)Math.Round((double)earned / maxScore * 100) : 0;
154 bool passed = pct >= passingScore;
155
156 return new QuizSubmitResponse(pct, passed, earned, maxScore, results);
157 }

159 // Рекурсивно парсит correct: число, строка, массив строк/чисел, массив массивов, null
160 private static object? ParseCorrect(JsonElement el)
161 {
162 return el.ValueKind switch
163 {
164 JsonValueKind.Null => null,
165 JsonValueKind.Number => el.GetInt32(),
166 JsonValueKind.String => el.GetString(),
167 JsonValueKind.Array => el.EnumerateArray()
168 .Select(ParseCorrect)
169 .ToList(),
170 _ => null
171 };
172 }

174 private string? FindTestFile(string quizId)
175 {
176 if (!Directory.Exists(_testsBasePath)) return null;
177
178 // First try exact filename match (fast path)
179 var byName = Directory.EnumerateFiles(_testsBasePath, $"{quizId}.json", SearchOption.AllDirectories)
180 .FirstOrDefault();
181 if (byName != null) return byName;
182
183 // Fallback: search by quizId field inside JSON
184 return Directory.EnumerateFiles(_testsBasePath, "*.json", SearchOption.AllDirectories)
185 .FirstOrDefault(file =>
186 {
187 try
188 {
189 using var doc = JsonDocument.Parse(File.ReadAllText(file));
190 return doc.RootElement.TryGetProperty("quizId", out var qid) && qid.GetString() == quizId;
191 }
192 catch { return false; }
193 });
194 }
