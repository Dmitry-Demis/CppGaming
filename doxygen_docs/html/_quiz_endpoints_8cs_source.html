15 // GET /api/quiz/list — список всех тестов (id + title + type)
16 group.MapGet("list", (IWebHostEnvironment env) =>
17 {
18 var testsRoot = Path.GetFullPath(Path.Combine(env.ContentRootPath, "tests"));
19 if (!Directory.Exists(testsRoot)) return Results.Ok(Array.Empty<object>());
20
21 var list = Directory.EnumerateFiles(testsRoot, "*.json", SearchOption.AllDirectories)
22 .Select(file =>
23 {
24 try
25 {
26 using var doc = System.Text.Json.JsonDocument.Parse(File.ReadAllText(file));
27 var root = doc.RootElement;
28 var id = root.TryGetProperty("quizId", out var qid) ? qid.GetString()
29 : root.TryGetProperty("id", out var rid) ? rid.GetString()
30 : Path.GetFileNameWithoutExtension(file);
31 var title = root.TryGetProperty("title", out var t) ? t.GetString() : id;
32 var type = root.TryGetProperty("type", out var tp) ? tp.GetString() : "mini";
33 var pick = root.TryGetProperty("pick", out var pk) ? pk.GetInt32() : 0;
34 var total = root.TryGetProperty("questions", out var qs) && qs.ValueKind == System.Text.Json.JsonValueKind.Array
35 ? qs.GetArrayLength() : 0;
36 return new { id, title, type, pick, total };
37 }
38 catch { return null; }
39 })
40 .Where(x => x != null)
41 .OrderBy(x => x!.id)
42 .ToList();
43
44 return Results.Ok(list);
45 });

99 // GET /api/quiz/{quizId}/question-stats/{questionId} — % правильных ответов на вопрос
100 group.MapGet("{quizId}/question-stats/{questionId:int}", async (
101 string quizId,
102 int questionId,
103 AppDbContext db) =>
104 {
105 var attempts = await db.TestAttempts
106 .Where(a => a.TestId == quizId)
107 .Select(a => new { a.CorrectQuestionIds, a.WrongQuestionIds })
108 .ToListAsync();
109
110 int total = 0, correct = 0;
111 foreach (var a in attempts)
112 {
113 bool wasCorrect = false, wasSeen = false;
114 if (!string.IsNullOrEmpty(a.CorrectQuestionIds))
115 {
116 try
117 {
118 var ids = System.Text.Json.JsonSerializer.Deserialize<List<int>>(a.CorrectQuestionIds);
119 if (ids?.Contains(questionId) == true) { wasCorrect = true; wasSeen = true; }
120 }
121 catch { }
122 }
123 if (!string.IsNullOrEmpty(a.WrongQuestionIds))
124 {
125 try
126 {
127 var ids = System.Text.Json.JsonSerializer.Deserialize<List<int>>(a.WrongQuestionIds);
128 if (ids?.Contains(questionId) == true) wasSeen = true;
129 }
130 catch { }
131 }
132 if (wasSeen) { total++; if (wasCorrect) correct++; }
133 }
134
135 if (total == 0) return Results.Ok(new { totalAttempts = 0, correctPct = 0 });
136 return Results.Ok(new { totalAttempts = total, correctPct = (int)Math.Round(correct * 100.0 / total) });
137 });

139 // GET /api/quiz/{quizId}/score-distribution — распределение результатов
140 group.MapGet("{quizId}/score-distribution", async (
141 string quizId,
142 ProfileService profileService,
143 AppDbContext db,
144 HttpContext ctx) =>
145 {
146 var isuNumber = ctx.Request.Headers["X-Isu-Number"].FirstOrDefault();
147
148 // Если передан query ?score=N — используем его (для экрана результатов сразу после теста)
149 int? overrideScore = null;
150 if (ctx.Request.Query.TryGetValue("score", out var sv) && int.TryParse(sv, out var parsedScore))
151 overrideScore = parsedScore;
152
153 int? myLast = overrideScore;
154 if (myLast == null && !string.IsNullOrEmpty(isuNumber))
155 {
156 var profile = await profileService.GetProfileAsync(isuNumber);
157 if (profile != null)
158 {
159 var myStats = await db.ParagraphTestStats
160 .Where(s => s.UserId == profile.Id && s.TestId == quizId)
161 .FirstOrDefaultAsync();
162 myLast = myStats?.LastScore;
163 }
164 }
165
166 // Берём последний результат каждого пользователя
167 var lastScores = await db.ParagraphTestStats
168 .Where(s => s.TestId == quizId && s.AttemptsCount > 0)
169 .Select(s => s.LastScore)
170 .ToListAsync();
171
172 if (lastScores.Count == 0)
173 return Results.Ok(new { totalAttempts = 0, pctAbove = 0, pctSame = 0, pctBelow = 0 });
174
175 int score = myLast ?? 0;
176 int above = lastScores.Count(s => s > score);
177 int same = lastScores.Count(s => s == score);
178 int below = lastScores.Count(s => s < score);
179 int total = lastScores.Count;
180
181 return Results.Ok(new
182 {
183 totalAttempts = total,
184 pctAbove = (int)Math.Round(above * 100.0 / total),
185 pctSame = (int)Math.Round(same * 100.0 / total),
186 pctBelow = (int)Math.Round(below * 100.0 / total),
187 myScore = score,
188 });
189 });

191 // GET /api/quiz/{quizId}/pick-questions — SR-выборка с учётом купленных стандартов
192 group.MapGet("{quizId}/pick-questions", async (
193 string quizId,
194 IQuizService quizService,
195 ProfileService profileService,
196 QuestionProgressService qpService,
197 IShopRepository shopRepo,
198 HttpContext ctx) =>

209 .Select(q => q.Id)

211
212 int pick = quiz.Pick > 0 ? Math.Min(quiz.Pick, availableIds.Count) : availableIds.Count;
213
214 if (string.IsNullOrEmpty(isuNumber))
215 {
216 var rng = new Random();
217 return Results.Ok(availableIds.OrderBy(_ => rng.Next()).Take(pick).ToList());
218 }
219
220 var profile = await profileService.GetProfileAsync(isuNumber);
221 if (profile is null)
222 {
223 var rng = new Random();
224 return Results.Ok(availableIds.OrderBy(_ => rng.Next()).Take(pick).ToList());
225 }
226
227 var picked = await qpService.PickQuestionIdsAsync(profile.Id, quizId, availableIds, pick);
228 return Results.Ok(picked);
229 });

232 // Возвращает Set купленных стандартов: {"11", "14", "17", ...}
233 // C++98 бесплатен всегда
234 private static async Task<HashSet<string>> GetUnlockedStds(
235 ProfileService profileService,
236 IShopRepository shopRepo,
237 HttpContext ctx)
238 {
239 var result = new HashSet<string> { "98" };
240 var isuNumber = ctx.Request.Headers["X-Isu-Number"].FirstOrDefault();
241 if (string.IsNullOrEmpty(isuNumber)) return result;
242
243 var profile = await profileService.GetProfileAsync(isuNumber);
244 if (profile is null) return result;
245
246 var purchases = await shopRepo.GetUserPurchasesAsync(profile.Id);
247 foreach (var p in purchases.Where(p => p.ItemId.StartsWith("std_cpp")))
248 result.Add(p.ItemId.Replace("std_cpp", ""));
249
250 return result;
251 }
