8public static class AnalyticsEndpoints
9{
10 public static void MapAnalyticsEndpoints(this IEndpointRouteBuilder app)
11 {
12 var group = app.MapGroup("/api/analytics");
13
14 // GET /api/analytics/{isuNumber}
15 // Returns all analytics data needed for the 7 dashboard charts
16 group.MapGet("{isuNumber}", async (
17 string isuNumber,
18 ProfileService profileService,
19 AppDbContext db) =>
20 {
21 var profile = await profileService.GetProfileAsync(isuNumber);
22 if (profile is null) return Results.NotFound();
23 int userId = profile.Id;
24
25 // ── 1. Activity heatmap: daily XP earned last 90 days ──────────
26 var since90 = DateTime.UtcNow.Date.AddDays(-89);
27 var activityLogs = await db.ActivityLogs
28 .Where(l => l.UserId == userId && l.CreatedAt >= since90)
29 .ToListAsync();
30
31 var heatmap = activityLogs
32 .GroupBy(l => l.CreatedAt.Date)
33 .Select(g => new { date = g.Key.ToString("yyyy-MM-dd"), xp = g.Sum(l => l.XpEarned), actions = g.Count() })
34 .OrderBy(x => x.date)
35 .ToList();
36
37 // ── 2. All test attempts (used in sections 2, 5, 9) ───────────
38 var allAttemptsList = await db.TestAttempts
39 .Where(a => a.UserId == userId)
40 .Select(a => new { a.TestId, a.ParagraphId, a.Score, a.CreatedAt })
41 .ToListAsync();
42
43 // Test scores over time (last 30 attempts)
44 var testTimeline = allAttemptsList
45 .OrderByDescending(a => a.CreatedAt)
46 .Take(30)
47 .OrderBy(a => a.CreatedAt)
48 .Select(a => new { date = a.CreatedAt.ToString("yyyy-MM-dd"), score = a.Score, testId = a.TestId, paragraphId = a.ParagraphId })
49 .ToList();
50
51 // ── 3. Mastery by paragraph (bar chart) ───────────────────────
52 var allTestStats = await db.ParagraphTestStats
53 .Where(t => t.UserId == userId && t.AttemptsCount > 0)
54 .ToListAsync();
55
56 var masteryByParagraph = allTestStats
57 .GroupBy(t => t.ParagraphId)
58 .Select(g => new
59 {
60 paragraphId = g.Key,
61 mastery = (int)Math.Round(g.Average(t => t.BestScore)),
62 attempts = g.Sum(t => t.AttemptsCount),
63 tests = g.Count()
64 })
65 .OrderByDescending(x => x.mastery)
66 .ToList();
67
68 // ── 4. Reading time by chapter (stacked bar) ──────────────────
69 var allReading = await db.ParagraphReadingStats
70 .Where(r => r.UserId == userId && r.TotalReadingSeconds > 0)
71 .ToListAsync();
72
73 // ── 5. Score distribution (histogram: 0-9, 10-19, ..., 90-100) ─
74 var allAttemptScores = allAttemptsList.Select(a => a.Score).ToList();
75
76 var scoreDistribution = Enumerable.Range(0, 10).Select(bucket => new
77 {
78 label = bucket == 9 ? "90-100" : $"{bucket * 10}-{bucket * 10 + 9}",
79 count = allAttemptScores.Count(s => bucket == 9 ? s >= 90 : s >= bucket * 10 && s < (bucket + 1) * 10)
80 }).ToList();
81
82 // ── 6. Pass/fail ratio per paragraph ──────────────────────────
83 var passFailByParagraph = allTestStats
84 .GroupBy(t => t.ParagraphId)
85 .Select(g => new
86 {
87 paragraphId = g.Key,
88 passed = g.Count(t => t.BestScore >= 70),
89 failed = g.Count(t => t.BestScore < 70)
90 })
91 .ToList();
92
93 // ── 7. XP earned by action type (pie chart) ───────────────────
94 var xpByAction = activityLogs
95 .GroupBy(l => l.ActionType)
96 .Select(g => new { action = g.Key, xp = g.Sum(l => l.XpEarned), count = g.Count() })
97 .OrderByDescending(x => x.xp)
98 .ToList();
99
100 // ── 8. Weekly activity (last 12 weeks) ────────────────────────
101 var since12w = DateTime.UtcNow.Date.AddDays(-83); // 12 * 7
102 var weeklyLogs = await db.ActivityLogs
103 .Where(l => l.UserId == userId && l.CreatedAt >= since12w)
104 .ToListAsync();
105
106 var weeklyActivity = weeklyLogs
107 .GroupBy(l => {
108 var d = l.CreatedAt.Date;
109 // ISO week start (Monday)
110 int diff = (7 + (int)d.DayOfWeek - (int)DayOfWeek.Monday) % 7;
111 return d.AddDays(-diff).ToString("yyyy-MM-dd");
112 })
113 .Select(g => new
114 {
115 weekStart = g.Key,
116 xp = g.Sum(l => l.XpEarned),
117 tests = g.Count(l => l.ActionType == "test_complete"),
118 reads = g.Count(l => l.ActionType == "read_paragraph"),
119 codeRuns = g.Count(l => l.ActionType == "code_run")
120 })
121 .OrderBy(x => x.weekStart)
122 .ToList();
123
124 // ── 9. Improvement trend: first vs best score per test ─────────
125 var firstAttempts = allAttemptsList
126 .GroupBy(a => a.TestId)
127 .Select(g => new { testId = g.Key, firstScore = g.OrderBy(a => a.CreatedAt).First().Score })
128 .ToList();
129
130 var improvementData = allTestStats
131 .Join(firstAttempts, ts => ts.TestId, fa => fa.testId,
132 (ts, fa) => new
133 {
134 testId = ts.TestId,
135 paragraphId = ts.ParagraphId,
136 firstScore = fa.firstScore,
137 bestScore = ts.BestScore,
138 improvement = ts.BestScore - fa.firstScore
139 })
140 .Where(x => x.improvement > 0)
141 .OrderByDescending(x => x.improvement)
142 .Take(10)
143 .ToList();
144
145 return Results.Ok(new
146 {
147 heatmap,
148 testTimeline,
149 masteryByParagraph,
150 readingStats = allReading.Select(r => new
151 {
152 r.ParagraphId,
153 r.TotalReadingSeconds,
154 r.ReadingSessionsCount,
155 r.CodeRunsCount
156 }),
157 scoreDistribution,
158 passFailByParagraph,
159 xpByAction,
160 weeklyActivity,
161 improvementData,
162 totalAttempts = allAttemptScores.Count,
163 avgScore = allAttemptScores.Count > 0 ? (int)Math.Round(allAttemptScores.Average()) : 0,
164 passRate = allAttemptScores.Count > 0
165 ? (int)Math.Round(allAttemptScores.Count(s => s >= 70) * 100.0 / allAttemptScores.Count)
166 : 0
167 });
168 });
169 }
