1// Переменные и типы данных
2// Переменная — именованная область памяти для хранения значения.
3// Каждая переменная имеет тип, определяющий, что именно в ней хранится и какие операции допустимы.
4
5#include <iostream>
6
7int main()
8{
9 // Объявление переменных без инициализации (содержат мусор, пока не присвоим значение)
10 int age{}; // Целое число
11 double price{}; // Вещественное число
12 char letter{}; // Один символ
13 bool isReady{}; // Логическое значение (true/false)
14
15 // Рекомендуется объявлять переменные по одной для лучшей читаемости.
16 // Избегайте стиля: int x, y, z; — это усложняет поддержку кода.
17 int x{};
18 int y{};
19 int z{};
20
21 // Объявление с инициализацией (предпочтительный способ)
22 int count{ 10 };
23 std::cout << "count = " << count << '\n';
24
25 // Тип определяет размер в памяти (в байтах) и набор допустимых операций.
26 // Оператор sizeof возвращает размер типа в байтах во время компиляции.
27 std::cout << "Размер int: " << sizeof(int) << " байт\n";
28 std::cout << "Размер double: " << sizeof(double) << " байт\n";
29
30 return 0;
31}
