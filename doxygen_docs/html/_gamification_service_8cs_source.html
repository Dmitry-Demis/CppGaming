72 // Called on login to check if streak was broken (no activity yesterday and no freeze)
73 public async Task<StreakCheckResult> CheckStreakOnLoginAsync(int userId)
74 {
75 var profile = await repo.GetAsync(userId);
76 if (profile is null) return new StreakCheckResult(0, 0, 0, false, false, 0, 0);
77
78 var today = DateTime.UtcNow.Date;
79 var lastDate = profile.LastActivityDate.Date;
80 bool usedFreeze = false;
81 bool streakBroken = false;
82 int coinsReward = 0;
83 int xpReward = 0;
84
85 if (lastDate == today)
86 {
87 // Already checked today — just return current state, no rewards
88 return new StreakCheckResult(
89 profile.CurrentStreak, profile.FreezeCount, profile.SnowflakeCount,
90 false, false, 0, 0);
91 }
92
93 if (lastDate == today.AddDays(-1))
94 {
95 // Consecutive day — increment streak and give daily reward
96 profile.CurrentStreak++;
97 profile.TotalStreakDays++;
98 profile.MaxStreak = Math.Max(profile.MaxStreak, profile.CurrentStreak);
99 profile.LastActivityDate = DateTime.UtcNow;
100 AwardFreezesIfDue(profile);
101
102 // Daily login reward
103 bool isMilestoneDay = profile.CurrentStreak % 7 == 0;
104 coinsReward = isMilestoneDay ? 100 : 50;
105 xpReward = isMilestoneDay ? 50 : 10;
106 profile.Coins += coinsReward;
107 profile.Xp += xpReward;
108 profile.Level = CalcLevel(profile.Xp);
109 }
110 else if (lastDate < today.AddDays(-1))
111 {
112 // Missed day(s)
113 if (profile.FreezeCount > 0)
114 {
115 profile.FreezeCount--;
116 profile.SnowflakeCount++;
117 usedFreeze = true;
118 profile.LastActivityDate = DateTime.UtcNow;
119 // Still give daily reward (freeze saved the streak)
120 bool isMilestoneDay = profile.CurrentStreak % 7 == 0;
121 coinsReward = isMilestoneDay ? 100 : 50;
122 xpReward = isMilestoneDay ? 50 : 10;
123 profile.Coins += coinsReward;
124 profile.Xp += xpReward;
125 profile.Level = CalcLevel(profile.Xp);
126 }
127 else
128 {
129 streakBroken = profile.CurrentStreak > 1;
130 profile.CurrentStreak = 1;
131 profile.TotalStreakDays++;
132 profile.LastActivityDate = DateTime.UtcNow;
133 // Small consolation reward on restart
134 coinsReward = 50;
135 xpReward = 10;
136 profile.Coins += coinsReward;
137 profile.Xp += xpReward;
138 profile.Level = CalcLevel(profile.Xp);
139 }
140 }
141
142 await repo.SaveAsync(profile);
143
144 return new StreakCheckResult(
145 profile.CurrentStreak,
146 profile.FreezeCount,
147 profile.SnowflakeCount,
148 usedFreeze,
149 streakBroken,
150 coinsReward,
151 xpReward
152 );
153 }

190 public async Task<bool> UnlockSlotAsync(int userId, string slotKey, int price)
191 {
192 var profile = await repo.GetAsync(userId);
193 if (profile is null) return false;
194
195 var unlocked = (profile.UnlockedSlots ?? "")
196 .Split(',', StringSplitOptions.RemoveEmptyEntries)
197 .ToHashSet();
198
199 if (unlocked.Contains(slotKey)) return false; // already unlocked
200
201 if (profile.Coins < price) return false; // not enough coins
202
203 profile.Coins -= price;
204 unlocked.Add(slotKey);
205 profile.UnlockedSlots = string.Join(',', unlocked);
206 await repo.SaveAsync(profile);
207 return true;
208 }
